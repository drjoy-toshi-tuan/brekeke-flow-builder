### W-001: 当日判定スクリプトの月1桁フォーマット非対応

  // 追加: 月・日の1桁対応
  var dateMatch = inputStr.match(/^(\d{4})-(\d{1,2})-(\d{1,2})/);
  if (dateMatch) {
    finalInputDate = dateMatch[1] + "-" +
      ("0" + dateMatch[2]).slice(-2) + "-" +
      ("0" + dateMatch[3]).slice(-2);
