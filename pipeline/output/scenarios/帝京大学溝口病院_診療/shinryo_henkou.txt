=== OpenAI_診療科_変更 ===
  [timeout] condition=^TIMEOUT$ -> リトライ_診療科_変更
  [error] condition=^ERROR$ -> リトライ_診療科_変更
  [no_result] condition=^NO_RESULT$ -> リトライ_診療科_変更
  [精神科] condition=^精神科$ -> 終話_直接来院_アナウンス
  [心療内科] condition=^心療内科$ -> 終話_直接来院_アナウンス
  [歯科] condition=^歯科$ -> 終話_予約不可_アナウンス
  [口腔外科] condition=^口腔外科$ -> 終話_予約不可_アナウンス
  [放射線科] condition=^放射線科$ -> 代表電話_アナウンス_放射線
  [補聴器外来] condition=^補聴器外来$ -> 代表電話_アナウンス_放射線
  [リストにない] condition=^リストにない$ -> リトライ_診療科_変更
  [その他] condition=^.*$ -> 入力_検査有無確認

=== リトライ_注射装具_変更 ===
  [Retry] condition=true -> 入力_注射装具確認_変更
  [No more] condition=false -> 入力_診療科_聴取_変更

=== リトライ_注射装具_キャンセル ===
  [Retry] condition=true -> 入力_注射装具確認_キャンセル
  [No more] condition=false -> 入力_診療科_聴取_キャンセル
